# upload file to specific folder
def upload_file_to_specific_folder(folder_id, file_name):
    file_metadata = {'title':file_name, 'parents': [{'id': folder_id, 'kind': 'drive#childList'}]}
    folder = drive.CreateFile(file_metadata)
    folder.SetContentFile(file_name)
    folder.Upload()

# create folder within a folder
def createFolder(folderName, parentID = None):
       # Create a folder on Drive, returns the newely created folders ID
       body = {
         'title': folderName,
         'mimeType': "application/vnd.google-apps.folder"
       }
       if parentID:
           body['parents'] = [{'id': parentID}]
       root_folder = drive_service.files().insert(body = body).execute()
       return root_folder['id']
